**_bottom_to_top(session):**
```lua
local function _bottom_to_top(session)
  local order = {}
  for i = 1, #session.cursors do
    if not session.cursors[i]._stopped then
      order[#order + 1] = i
    end
  end
  table.sort(order, function(a, b)
    local ra, ca = session.cursors[a]:pos()
    local rb, cb = session.cursors[b]:pos()
    if ra ~= rb then return ra > rb end
    return ca > cb
  end)
  return order
end
```

**M.exec(session, keys):**
Implements the locked CONTEXT.md design exactly:
1. Guard: return early if session._stopped or #session.cursors == 0
2. Encode termcodes once: `vim.api.nvim_replace_termcodes(keys, true, false, true)`
3. Save vim.o.eventignore
4. pcall wrapper around the entire loop + undo block:
   - Set vim.o.eventignore = 'all'
   - Call undo.begin_block(session)
   - For each idx in _bottom_to_top(session):
     - if not r._stopped: read row, col from r:pos() (0-indexed)
     - pcall(nvim_win_set_cursor, 0, {row+1, col})  -- 1-indexed row (PITFALL-04)
     - pcall(nvim_feedkeys, encoded, 'x', false)    -- 'x' mode (PITFALL-03)
   - Call undo.end_block(session)
5. ALWAYS restore vim.o.eventignore (PITFALL-02 prevention) — outside the pcall
6. On pcall error: vim.notify with WARN level
7. Store keys in session._vm_dot
8. Call hl.redraw(session)

**M.yank(session):**
Per-cursor yank without the general undo block (yank is not undoable in standard Vim):
1. Guard: return if session._stopped or empty
2. Save/restore vim.o.eventignore around the loop
3. For each idx in _bottom_to_top(session):
   - row, col = r:pos()
   - nvim_win_set_cursor(0, {row+1, col})
   - vim.fn.setreg('"', '')  -- clear to detect yank result
   - pcall(nvim_feedkeys, nvim_replace_termcodes('yiw', true, false, true), 'x', false)
   - session._vm_register = session._vm_register or {}
   - session._vm_register[idx] = { text = vim.fn.getreg('"'), type = vim.fn.getregtype('"') }
4. Store 'yiw' in session._vm_dot
5. hl.redraw(session)

Note: yiw is the default yank motion (word under cursor). This matches the VimScript source behavior for bare `y`. Phase 6 adds operator-pending capture for `yw`, `y3j`, etc.

**M.paste(session, before):**
1. Guard: return if session._stopped or empty
2. use_vm_reg = session._vm_register and #session._vm_register > 0
3. Save vim.o.eventignore; set to 'all' in pcall wrapper
4. undo.begin_block(session)
5. For each idx in _bottom_to_top(session):
   - row, col = r:pos(); nvim_win_set_cursor(0, {row+1, col})
   - If use_vm_reg and session._vm_register[idx] exists:
     - setreg('"', entry.text, entry.type)
   - keys = before and 'P' or 'p'
   - pcall(nvim_feedkeys, nvim_replace_termcodes(keys, true, false, true), 'x', false)
6. undo.end_block(session)
7. Restore vim.o.eventignore (always, outside pcall)
8. session._vm_dot = before and 'P' or 'p'
9. hl.redraw(session)

**Category U: Undo grouping (FEAT-06)**

```lua
T['single undo after exec with 3 cursors creates exactly 1 undo entry'] = function()
  vim.api.nvim_buf_set_lines(buf, 0, -1, false, { 'aa', 'bb', 'cc' })
  local s = session_mod.start(buf, false)
  s.cursors[1] = region_mod.new(buf, 0, 0)
  s.cursors[2] = region_mod.new(buf, 1, 0)
  s.cursors[3] = region_mod.new(buf, 2, 0)
  s.primary_idx = 3
  local before = vim.fn.undotree().seq_cur
  edit.exec(s, 'dd')
  local after = vim.fn.undotree().seq_cur
  MiniTest.expect.equality(after - before, 1)
end

T['undo after exec reverses all cursor deletions'] = function()
  local s = session_mod.start(buf, false)
  s.cursors[1] = region_mod.new(buf, 0, 0)
  s.cursors[2] = region_mod.new(buf, 1, 0)
  s.primary_idx = 2
  edit.exec(s, 'dw')
  vim.api.nvim_set_current_buf(buf)
  vim.cmd('silent undo')
  local lines = vim.api.nvim_buf_get_lines(buf, 0, -1, false)
  MiniTest.expect.equality(lines[1], 'hello world')
  MiniTest.expect.equality(lines[2], 'foo bar baz')
end

T['no-op exec (cursor at EOF) does not advance undo sequence'] = function()
  vim.api.nvim_buf_set_lines(buf, 0, -1, false, { '' })
  local s = session_mod.start(buf, false)
  s.cursors[1] = region_mod.new(buf, 0, 0)
  s.primary_idx = 1
  local before = vim.fn.undotree().seq_cur
  edit.exec(s, 'dw')  -- nothing to delete on empty line
  local after = vim.fn.undotree().seq_cur
  MiniTest.expect.equality(after, before)  -- BUG-04: no spurious entry
end
```
