```lua
--- vim-visual-multi undo grouping
--- Wraps multi-cursor edits in atomic undo blocks.
--- Bug guard references:
---   BUG-02: Test buffers must be (false, false) — not scratch — for undo to work
---   BUG-03: Use vim.bo[buf].undolevels, NOT vim.o.undolevels
---   BUG-04: Short-circuit end_block if content unchanged to avoid spurious undo entry

local M = {}

--- Begin an undo block for a session.
--- Records the current undo sequence number and buffer content snapshot.
---@param session table  Must have `.buf` (bufnr integer)
function M.begin_block(session)
  vim.api.nvim_buf_call(session.buf, function()
    session._undo_seq_before = vim.fn.undotree().seq_cur
    session._undo_lines_before = vim.api.nvim_buf_get_lines(
      session.buf, 0, -1, false
    )
  end)
end

--- End an undo block for a session.
--- BUG-04: Short-circuits if content is unchanged — avoids advancing seq_cur
--- on no-op operations (e.g., search with no results).
---@param session table
function M.end_block(session)
  local lines_after = vim.api.nvim_buf_get_lines(session.buf, 0, -1, false)
  local util = require('visual-multi.util')
  if util.deep_equal(session._undo_lines_before, lines_after) then
    -- No content change — clear state, do NOT create undo entry
    session._undo_seq_before = nil
    session._undo_lines_before = nil
    return
  end
  vim.api.nvim_buf_call(session.buf, function()
    session._undo_seq_after = vim.fn.undotree().seq_cur
  end)
  session._undo_lines_before = nil
end

--- Convenience wrapper: calls begin_block, runs fn(session), then end_block.
--- Returns the return value of fn.
---@param session table
---@param fn function
---@return any
function M.with_undo_block(session, fn)
  M.begin_block(session)
  local result = fn(session)
  M.end_block(session)
  return result
end

--- Flush the undo history of a buffer into a single entry.
--- Used to consolidate all per-cursor edits into one undoable unit.
--- BUG-03: Uses vim.bo[buf].undolevels (buffer-local), NOT vim.o.undolevels (global).
---@param buf integer
function M.flush_undo_history(buf)
  -- BUG-03: buffer-local undolevels, never vim.o.undolevels
  local saved = vim.bo[buf].undolevels
  vim.bo[buf].undolevels = -1
  -- Apply a no-visible-change to force undo history flush
  vim.api.nvim_buf_call(buf, function()
    vim.cmd('silent! undojoin')
  end)
  vim.bo[buf].undolevels = saved
end

return M
```

**Design note on `flush_undo_history`:** The `undojoin` approach is cleaner than the `nvim_buf_set_lines` no-op change. Either works but `undojoin` avoids modifying buffer content. If `undojoin` raises in certain contexts (e.g., after a redo), wrap in `pcall`. Add a comment in the code explaining the choice.

**Category B: begin_block and end_block behavior**

```lua
T['begin_block records seq_before'] = function()
  -- buf setup omitted for brevity; use the before_each helper
  local session = { buf = buf, _stopped = false }
  require('visual-multi.undo').begin_block(session)
  MiniTest.expect.equality(type(session._undo_seq_before), 'number')
end

T['end_block clears state on no-change (BUG-04)'] = function()
  local session = { buf = buf, _stopped = false }
  local undo = require('visual-multi.undo')
  undo.begin_block(session)
  -- Make NO changes to the buffer
  undo.end_block(session)
  -- State must be cleared (short-circuit path)
  MiniTest.expect.equality(session._undo_seq_before, nil)
  MiniTest.expect.equality(session._undo_lines_before, nil)
  -- seq_cur must not have advanced (no spurious undo entry)
  local seq_after = vim.api.nvim_buf_call(buf, function()
    return vim.fn.undotree().seq_cur
  end)
  -- seq_after should equal whatever it was before (or 0) — not advanced
  -- We just confirm end_block did not create an undo entry by checking _undo_seq_after is nil
  MiniTest.expect.equality(session._undo_seq_after, nil)
end

T['end_block records seq_after on real change'] = function()
  local session = { buf = buf, _stopped = false }
  local undo = require('visual-multi.undo')
  undo.begin_block(session)
  -- Make a real change
  vim.api.nvim_buf_set_lines(buf, 0, 1, false, {'changed line'})
  undo.end_block(session)
  MiniTest.expect.equality(type(session._undo_seq_after), 'number')
  MiniTest.expect.equality(session._undo_lines_before, nil)  -- cleared
end
```

**Category C: with_undo_block**

```lua
T['with_undo_block wraps begin and end'] = function()

  undo.with_undo_block(session, function(s)

  end)
  MiniTest.expect.equality(called, true)
  -- _undo_lines_before should be cleared by end_block

end
```

**Standard before_each/after_each:**
```lua
local buf
T['before_each'] = function()
  buf = vim.api.nvim_create_buf(false, false)
  vim.bo[buf].buftype = 'nofile'
  vim.bo[buf].bufhidden = 'wipe'

  pcall(vim.api.nvim_buf_delete, buf, {force=true})
end
```
