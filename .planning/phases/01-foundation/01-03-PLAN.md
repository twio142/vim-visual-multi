**`M.define_groups()`** — Define all four VM highlight groups using `default = true` so user colorschemes override them without errors:
```lua
function M.define_groups()
  vim.api.nvim_set_hl(0, 'VMCursor', { default = true, link = 'Cursor' })
  vim.api.nvim_set_hl(0, 'VMExtend', { default = true, link = 'Visual' })
  vim.api.nvim_set_hl(0, 'VMInsert', { default = true, link = 'DiffChange' })
  vim.api.nvim_set_hl(0, 'VMSearch', { default = true, link = 'Search' })
end
```

**`M.draw_cursor(buf, row, col, mark_id)`** — Places or updates a cursor extmark. `mark_id` is nil on first call (returns new id); pass the returned id on subsequent calls for in-place update:
```lua
function M.draw_cursor(buf, row, col, mark_id)
  return vim.api.nvim_buf_set_extmark(buf, M.ns, row, col, {
    id       = mark_id,   -- nil = create new; integer = update in place
    end_row  = row,
    end_col  = col + 1,
    hl_group = 'VMCursor',
    priority = 200,
    hl_mode  = 'combine',
    strict   = false,
  })
end
```

**`M.draw_selection(buf, row, scol, ecol, mark_id)`** — Places or updates an extend-mode selection extmark spanning `scol` to `ecol`:
```lua
function M.draw_selection(buf, row, scol, ecol, mark_id)
  return vim.api.nvim_buf_set_extmark(buf, M.ns, row, scol, {
    id       = mark_id,
    end_row  = row,
    end_col  = ecol,
    hl_group = 'VMExtend',
    priority = 200,
    hl_mode  = 'combine',
    strict   = false,
  })
end
```

**`M.clear(session_or_buf)`** — Clear all VM extmarks from a buffer. Uses `util.is_session` dispatch (BUG-05 pattern — single function, no duplicate):
```lua
function M.clear(session_or_buf)
  local util = require('visual-multi.util')
  local buf = util.is_session(session_or_buf) and session_or_buf.buf
              or session_or_buf
  vim.api.nvim_buf_clear_namespace(buf, M.ns, 0, -1)
end
```

**`M.clear_region(buf, mark_id)`** — Delete a single extmark by id:
```lua

end
```

Create `lua/visual-multi/region.lua`. This is a Tier-1 module that depends on `highlight.lua` (for `M.ns`) and optionally `util.lua` (for `is_session` if needed in future). Do NOT create any circular back-reference.

Implementation uses the metatable (OOP) pattern. A Region is created with `Region.new(buf, row, col)`, updated with `Region:move(row, col)`, queried with `Region:pos()`, and removed with `Region:remove()`.

```lua
local M = {}
local Region = {}
Region.__index = Region

--- Create a new cursor-mode region at (row, col) in buf.
--- row, col are 0-indexed (Neovim API convention).
---@return Region
function Region.new(buf, row, col)
  local hl = require('visual-multi.highlight')
  local r = setmetatable({ buf = buf, _stopped = false }, Region)
  r.mark_id = vim.api.nvim_buf_set_extmark(buf, hl.ns, row, col, {
    end_row  = row,
    end_col  = col + 1,
    hl_group = 'VMCursor',
    priority = 200,
    hl_mode  = 'combine',
    strict   = false,
  })
  return r
end

--- Return current extmark position as 0-indexed (row, col).
---@return integer row, integer col
function Region:pos()
  local hl = require('visual-multi.highlight')
  local info = vim.api.nvim_buf_get_extmark_by_id(
    self.buf, hl.ns, self.mark_id, {}
  )
  return info[1], info[2]
end

--- Move the extmark to a new (row, col) in place (same mark_id).
--- Uses the `id=` parameter for O(log n) update — no clear-and-recreate.
function Region:move(row, col)
  local hl = require('visual-multi.highlight')
  vim.api.nvim_buf_set_extmark(self.buf, hl.ns, row, col, {
    id       = self.mark_id,
    end_row  = row,
    end_col  = col + 1,
    hl_group = 'VMCursor',
    priority = 200,
    hl_mode  = 'combine',
    strict   = false,
  })
end

--- Remove this region's extmark from the buffer and mark as stopped.
function Region:remove()
  local hl = require('visual-multi.highlight')
  pcall(vim.api.nvim_buf_del_extmark, self.buf, hl.ns, self.mark_id)
  self._stopped = true
end

M.new = Region.new
return M
```
